class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        if nums==[]: return 0
        i,j=0,len(nums)-1
        while i<j:
            if nums[i]==val:
                nums[i],nums[j]=nums[j],nums[i]
                j-=1
            else:
                i+=1
        if nums[i]==val: i-=1
        return i+1
